// Shorten the given string to a maximum of maxChars, including
// the "..." shortening indicator. MaxChars must be greater than 3.
function shortenString(string, maxChars = 60) {
    let shortened = string || ``;
    if (maxChars > 3 && shortened.length > maxChars - 3) {
        shortened = `${shortened.substring(0, maxChars - 3)}...`;
    }
    return shortened;
}
// Simplify and collapse whitespace from the given string:
// - replace all whitespace (e.g., newlines, &nbsp;) with blanks
// - replace sequences of blanks with a single blank
// Returns the converted string
function sanitizeWhitespace(string) {
    let sanitized = string || ``;
    sanitized = sanitized.replace(/\s/g, ` `);
    sanitized = sanitized.replace(/  +/g, ` `);
    return sanitized;
}
// Remove or convert extra whitespace that is located outside of pre-formatted blocks:
// - Tokenize by <pre> and </pre>
// - Simplify and collapse whitespace from the parts that are outside outside of <pre>...</pre> elements
// - Supports nested <pre> elements, like "<pre>abc<pre>def</pre>ghi</pre>"
//   as a single preformatted block
// - If a starting <pre> is not matched by a closing </pre>, then the complete text after
//   the starting <pre> is treated as pre-formatted text
function sanitizeStringOutsidePreformatted(string) {
    const tokenArray = string.split(/(<[/]?pre>)/g);
    let preformatLevel = 0;
    const sanitizedArray = [];
    for (let i = 0; i < tokenArray.length; i++) {
        const token = tokenArray[i];
        if (token === `<pre>`) {
            preformatLevel++;
            sanitizedArray.push(token);
        }
        else if (token === `</pre>`) {
            preformatLevel--;
            sanitizedArray.push(token);
        }
        else if (preformatLevel === 0) {
            sanitizedArray.push(sanitizeWhitespace(token));
        }
        else {
            sanitizedArray.push(token);
        }
    }
    return sanitizedArray.join(``);
}
// Sanitize the string:
//  - trim the string
//  - reduce whitespace outside of preformatted (<pre>) blocks
//  - finally, insert those newlines again that are actually required for preformatted blocks
//    and have been replaced with <CODE_BLOCK_NEWLINE> substitutes before
function sanitizeString(string) {
    let sanitized = string || ``;
    sanitized = sanitized.trim();
    sanitized = sanitizeStringOutsidePreformatted(sanitized);
    sanitized = sanitized.replace(/<CODE_BLOCK_NEWLINE>/g, `\n`);
    return sanitized;
}
module.exports = {
    shortenString,
    sanitizeString
};
